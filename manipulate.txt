#include<fstream.h>

void main()
{
ofstream outfile;
ifstream infile;
char source [10],target[10];
char ch;
clrscr();
cout<<"enter the source file name :";

cout<<"enter the target file name :";
cin>>target;
infile.open(source);

}

}
